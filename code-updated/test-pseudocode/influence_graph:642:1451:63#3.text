1. IF water is safe for frog to lay eggs
2.     FOR each female frog
3.         LAY thousands of eggs
4.     END FOR
5. ELSE IF water is too dangerous
6.     PRINT "Water is too dangerous for frogs to lay eggs"
7. END IF
8. WHILE there are still egg clusters
9.     FLOAT egg clusters in water
10.     FOR each egg cluster
11.         HATCH tadpoles with gills
12.     END FOR
13. END WHILE
14. WHILE there are still tadpoles
15.     SWIM tadpoles in water
16.     FOR each tadpole
17.         IF tadpole has grown legs
18.             ABSORB tail
19.             BECOME froglet
20.         END IF
21.     END FOR
22. END WHILE
23. FOR each froglet
24.     BECOME frog
25. END FOR
